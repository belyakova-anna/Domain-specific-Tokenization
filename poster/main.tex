\begin{block}{Motivation}
\justifying
\begin{itemize}
    \item Biomedical text contains long compounds, abbreviations, and rare terms that are hard for generic tokenizers.
    \item Over-fragmentation increases sequence length, weakens entity boundaries, and may harm minority classes.
    \item Research question: does domain-specific tokenization improve both efficiency (fewer subtokens) and task robustness (classification + NER alignment)?
\end{itemize}
\vspace{0.15em}
\textbf{Main contributions}
\begin{itemize}
    \item We train two biomedical BPE tokenizers from PubMedQA text (\texttt{30k}, \texttt{50k requested / 39,681 actual}) with a reproducible setup.
    \item We evaluate intrinsic, downstream, and entity-aware tokenization behavior in one unified protocol.
    \item We show consistent gains over strong baselines (BERT-base, BioBERT, GPT-2) under a fixed classifier pipeline.
\end{itemize}
\end{block}

\begin{alertblock}{Hypothesis}
\justifying
Domain-trained BPE will reduce fragmentation in biomedical text and improve task-level robustness, especially for entity-heavy inputs.
\end{alertblock}

\begin{alertblock}{Key Takeaways}
\begin{itemize}
  \item Best tokenizer overall: \textbf{BioBPE-50k}.
  \item Intrinsic compactness and extrinsic quality improve consistently.
  \item Gains are strongest on entity-sensitive biomedical text (lower fragmentation, better single-piece entity rate).
\end{itemize}
\end{alertblock}

\begin{block}{Practical Deployment Guidance}
\justifying
\small
\begin{itemize}
    \item \textbf{Default:} BioBPE-50k (fertility 1.301, top macro-F1 0.3718, top NER score 0.3049).
    \item \textbf{If constrained:} BioBPE-30k with near-identical quality (macro-F1 gap 0.0008).
    \item \textbf{Gain vs best baseline (GPT-2):} \(\sim\)12.8\% lower fertility and +0.0075 macro-F1.
    \item \textbf{Highest impact:} long biomedical contexts and multi-morpheme entities.
\end{itemize}
\vspace{0.15em}
\textbf{Adoption path:} train in-domain BPE, validate on classification + NER diagnostics, then fine-tune full models.
\end{block}
